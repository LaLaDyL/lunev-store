    const documentHeight = document.documentElement.scrollHeight;
    
    // –ü–æ–∫–∞–∑—ã–≤–∞–µ–º —Ñ—É—Ç–µ—Ä –∫–æ–≥–¥–∞ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å –ø—Ä–æ–∫—Ä—É—Ç–∏–ª –¥–æ –Ω–∏–∂–Ω–µ–π —á–∞—Å—Ç–∏ —Å—Ç—Ä–∞–Ω–∏—Ü—ã
